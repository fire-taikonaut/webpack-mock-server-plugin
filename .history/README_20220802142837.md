# webpack-mock-server-plugin

## example

```js

  // webpack.config.js
  const WebpackMockServerPlugin = reuqire('webpack-mock-server-plugin');
  const path = require('path');
  {
  ...
    plugins: [
      new WebpackMockServerPlugin({
        dir: path.resolve(process.cwd(), '/mock'), // mock目录
        port: '8080', // mock服务端口
      })
    ]
  };

```
